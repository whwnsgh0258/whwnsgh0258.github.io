  public String convertToDatabaseColumn(Role role) {
    return role != null ? role.getValue() : null;
  }

  @Override
  public Role convertToEntityAttribute(String dbData) {
    for (Role role : Role.values()) {
      if (role.getValue().equals(dbData)) {

}
